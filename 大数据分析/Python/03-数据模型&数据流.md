# 1 数据类型

```python
a = 1
b = 1.1
c = True
d = 3j
e = "abc"
f = ['1,' '2']
g = ('1', '2')
h = set()
i = {'1', '2'}
j = {}
k = {'guo': 'guo'}
m = bin(3)
n = oct(9)
o = int(11)
p = hex(17)

print(type(a), type(b), type(c), type(d))
print(type(e), type(f), type(g))
print(type(h), type(i))
print(type(j), type(k))
print(m, n, o, p)
```
